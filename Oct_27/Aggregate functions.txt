-- HAVING Clause : Used to check the condition on aggregate fields. 

What is the difference between WHERE clause and HAVING Clause 

WHERE Clause used to check the conditions on Table Field(s), as HAVING clause used to check the condition on Aggregate Field(s).

select job,sum(sal) from Emp
group by job;

Display info for which job total salary more than or equal to 5000

select job,sum(sal) from Emp
group by job
having sum(sal) >= 5000;

Display records in which job contains more than 1 employee 

select job,count(*) from Emp
group by job
having count(*) >= 1;

select city,count(*) from Agent
group by city
having count(*) > 1;

select job,sum(sal) 'total sal',
   avg(sal) 'avg sal',
   max(sal) 'max sal',
   min(sal) 'min sal',
   count(*) 'total employees' 
from Emp 
WHERE JOB IN('CLERK','MANAGER','PRSIDENT','ANALYST')
group by job
having count(*) > 1
ORDER BY JOB;
